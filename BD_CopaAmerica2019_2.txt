--Creación de tablas
CREATE TABLE PaisEquipo
(
	Id CHAR(3) NOT NULL PRIMARY KEY,
	Nombre VARCHAR(100) NOT NULL,
	Apodo VARCHAR(100) NOT NULL,
	Descripcion VARCHAR(800) NOT NULL,
	Foto VARCHAR(800) NOT NULL,
	Participaciones INT NOT NULL,
	Titulos INT NOT NULL,
	Subcampeonatos INT NOT NULL,
	MedallasOro INT NOT NULL,
	MedallasPlata INT NOT NULL,
	MedallasBronce INT NOT NULL,
	TitulosMundiales INT NOT NULL
)
GO

CREATE TABLE Camiseta
(
	Id INT NOT NULL IDENTITY(1,1) PRIMARY KEY,
	Pais VARCHAR(30) NOT NULL,
	Jugador VARCHAR(30) NOT NULL,
	Numero INT NOT NULL,
	Cantidad INT NOT NULL,
	Precio DECIMAL(15,2) NOT NULL,
	Descripcion VARCHAR(30) NOT NULL
)
GO

SELECT * FROM Camiseta
GO

CREATE TABLE Entrenador
(
	Id CHAR(3) NOT NULL PRIMARY KEY,
	Nombre VARCHAR(50) NOT NULL,
	Apodo VARCHAR(50) NOT NULL,
	Nacionalidad VARCHAR(50) NOT NULL,
	Titulos INT NOT NULL,
	Edad INT NOT NULL,
	IdPaisEquipo CHAR(3) NOT NULL FOREIGN KEY REFERENCES PaisEquipo
)

CREATE TABLE EquiposEntrenador
(
	Id CHAR(3) NOT NULL PRIMARY KEY,
	Nombre VARCHAR(50) NOT NULL,
	Pais VARCHAR(50) NOT NULL,
	Titulos INT NOT NULL,
	IdEntrenador CHAR(3) NOT NULL FOREIGN KEY REFERENCES Entrenador
)
GO

CREATE TABLE Jugador
(
	Id CHAR(4) NOT NULL PRIMARY KEY,
	Nombre VARCHAR(50) NOT NULL,
	ApePaterno VARCHAR(50) NOT NULL,
	ApeMaterno VARCHAR(50) NOT NULL,
	Edad INT NOT NULL,
	Apodo VARCHAR(50) NOT NULL,
	Participaciones INT NOT NULL,
	Titulos INT NOT NULL,
	SubCampeonatos INT NOT NULL,
	MedallasOro INT NOT NULL,
	MedallasPlata INT NOT NULL,
	MedallasBronce INT NOT NULL,
	GolesEnCopaAmerica INT NOT NULL,
	Asistencias INT NOT NULL,
	Autogoles INT NOT NULL,
	ClubActual VARCHAR(50) NOT NULL,
	GolesCarrera INT NOT NULL,
	NumCamisetaSeleccion INT NOT NULL,
	NumLesiones INT NOT NULL,
	TotalTarjetasRojas INT NOT NULL,
	TotalTarjetasAmarrilla INT NOT NULL,
	TarjetasRojasActual INT NOT NULL,
	TarjetasAmarrillasActual INT NOT NULL,
	MejorJugador INT NOT NULL,
	FechaNacimiento VARCHAR(50) NOT NULL,
	Foto VARCHAR(700) NOT NULL,
	IdPaisEquipo CHAR(3) NOT NULL FOREIGN KEY REFERENCES PaisEquipo,
	CapitanSeleccion CHAR(2) NOT NULL DEFAULT 'NO'
)
GO

CREATE TABLE Estadio
(
	Id CHAR(3) NOT NULL PRIMARY KEY,
	Nombre VARCHAR(50) NOT NULL,
	Apodo VARCHAR(50) NOT NULL,
	Descripcion VARCHAR(500) NOT NULL,
	Foto VARCHAR(250) NOT NULL,
	IdPaisEquipo CHAR(3) NOT NULL FOREIGN KEY REFERENCES PaisEquipo,
	PartidosJugadosActual INT NOT NULL,
	TotalPartidosCopaAmerica INT NOT NULL,
	NumFinales INT NOT NULL,
	Fundacion VARCHAR(40) NOT NULL,
	Capacidad VARCHAR(40) NOT NULL,
	Dimensiones VARCHAR(140) NOT NULL,
	Ubicacion VARCHAR(40) NOT NULL,
	Coordenadas VARCHAR(100) NOT NULL,
	Propietario VARCHAR(100) NOT NULL,
)
GO

CREATE TABLE PartidoFinal
(
	Id CHAR(3) NOT NULL PRIMARY KEY,
	IdEstadio CHAR(3) NOT NULL FOREIGN KEY REFERENCES Estadio,
	MejorJugador VARCHAR(20) NOT NULL,
	GolesEquipoUno INT NOT NULL,
	GolesEquipoDos INT NOT NULL,
	Ganador VARCHAR(40) NOT NULL,
	Perdedor VARCHAR(40) NOT NULL,
	NombreCopaAmerica VARCHAR(40) NOT NULL, 
	Asistencia INT NOT NULL,
	Arbitro VARCHAR(40) NOT NULL
)
GO
